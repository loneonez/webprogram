HTMLタグ復習用（意味だけ表示）

<!DOCTYPE html> // HTML5文書であることを宣言
<html lang="ja"> // HTML文書の開始、言語は日本語

<head> // 頭。設定やCSS, JS読み込み
  <meta charset="UTF-8"> // 文字コード指定
  <meta name="viewport" content="width=device-width, initial-scale=1.0"> // スマホ対応
  <title>ページタイトル</title> // ブラウザタブに表示
  <link rel="stylesheet" href="style.css"> // 外部CSS読み込み
</head>

<body> // 本文、画面に表示される内容

  <header> // ヘッダー、ページ上部情報
    <h1>見出し1</h1> // 最重要見出し
    <h2>見出し2</h2> // サブ見出し
  </header>

  <nav> // ナビゲーションメニュー
    <ul>
      <li><a href="#">リンク1</a></li>
      <li><a href="#">リンク2</a></li>
    </ul>
  </nav>

  <main> // メインコンテンツ
    <section> // 内容のまとまり
      <h2>セクション見出し</h2>
      <p>段落。文章を書く場所</p>
      <strong>強調</strong>, <em>イタリック</em>
      <small>小さい文字</small>, <mark>マークアップ文字</mark>
      <a href="#">リンク</a>
      <img src="image.jpg" alt="画像説明">
      <figure>
        <img src="figure.jpg" alt="図の説明">
        <figcaption>図のキャプション</figcaption>
      </figure>
    </section>

    <article> // 記事コンテンツ
      <h2>記事タイトル</h2>
      <p>記事本文</p>
    </article>

    <aside> // サイド情報や補足
      <p>補足情報や広告など</p>
    </aside>

    <details> // 折りたたみ
      <summary>クリックで開く内容</summary>
      <p>折りたたみ内容</p>
    </details>

    <dialog open> // ダイアログ（モーダル）
      <p>ダイアログ内容</p>
    </dialog>

    <progress value="70" max="100"></progress> // 進捗バー
    <meter value="0.6">60%</meter> // メーター表示

    <form> // 入力フォーム
      <label for="name">名前:</label>
      <input type="text" id="name" name="name" placeholder="名前を入力">
      <input type="number" name="age" min="0" max="120">
      <input type="email" name="email">
      <textarea name="message" placeholder="メッセージ"></textarea>
      <select name="option">
        <option value="A">選択肢A</option>
        <option value="B">選択肢B</option>
      </select>
      <input type="checkbox" name="check" checked> チェックボックス
      <input type="radio" name="radio"> ラジオボタン
      <button type="submit">送信</button>
    </form>

    <table> // 表
      <thead><tr><th>見出し1</th><th>見出し2</th></tr></thead>
      <tbody><tr><td>データ1</td><td>データ2</td></tr></tbody>
      <tfoot><tr><td>合計1</td><td>合計2</td></tr></tfoot>
    </table>

    <blockquote>引用文</blockquote>
    <code>コード表示</code>
    <pre>整形済みテキスト</pre>
    <abbr title="HyperText Markup Language">HTML</abbr> // 略語
    <cite>参考文献名</cite>
    <time datetime="2025-09-15">2025年9月15日</time>

    <span>インライン要素</span>
    <div>ブロック要素</div>

  </main>

  <footer> <p>© 2025 ゆうたの復習ページ</p> </footer>

</body>
</html>
